## 🔍 搜索算法

```typescript
function searchUsers(query: string, excludeIds: number[]): User[] {
  // 1. 空查询返回所有用户
  if (!query.trim()) {
    return allUsers.filter(u => !excludeIds.includes(u.id));
  }
  
  // 2. 模糊匹配
  const lowerQuery = query.toLowerCase();
  return allUsers
    .filter(user => 
      !excludeIds.includes(user.id) &&
      (user.name.toLowerCase().includes(lowerQuery) ||
       user.bio?.toLowerCase().includes(lowerQuery))
    )
    .slice(0, 20); // 限制结果数量
}
```
